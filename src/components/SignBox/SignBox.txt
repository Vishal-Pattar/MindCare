import React, { useState } from 'react';
import './SignBox.css';
import { Link } from 'react-router-dom';

const LoginBox = () => {
    const [email, setEmail] = useState('');
    const [password, setPassword] = useState('');
    const [isUserValid, setIsUserValid] = useState(true);

    const validateEmail = (email) => {
        const regex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
        return regex.test(email);
    };

    const handleLogin = () => {
        const emailIsValid = validateEmail(email);
        const passwordIsValid = password.length >= 6;

        if (!emailIsValid || !passwordIsValid) {
            setIsUserValid(false);
        }
        else {
            setIsUserValid(true);
        }
    };

    return (
        <>
            <div className='signbox__container'>
                <div className='signbox__title'>Please Login to Continue</div>
                <div className={`signbox__email signbox__input ${!isUserValid ? 'invalid' : ''}`}>
                    <label htmlFor="email">Email</label>
                    <input 
                        type='email' 
                        id='email' 
                        placeholder='Enter your email address'
                        value={email}
                        onChange={(e) => setEmail(e.target.value)} 
                        required
                    />
                    {isUserValid ? null : <div className='signbox__error'>! Invalid Credentials</div>}
                </div>
                <div className={`signbox__password signbox__input ${!isUserValid ? 'invalid' : ''}`}>
                    <label htmlFor="pass">Password</label>
                    <input 
                        type='password' 
                        id='pass' 
                        placeholder='Enter Password'
                        value={password}
                        onChange={(e) => setPassword(e.target.value)} 
                        required
                    />
                    {isUserValid ? null : <div className='signbox__error'>! Invalid Credentials</div>}
                </div>
                <button className='signbox__button' onClick={handleLogin}>Login</button>
            </div>
            <div className='signbox__footer'>No account yet? <Link to='/register'>Register</Link></div>
            {isUserValid ? null : <div className='signbox__errorbox'>Lorem ipsum dolor sit amet consectetur adipisicing elit. Quia at beatae facere dolorum harum, molestiae maiores dolores ullam quibusdam debitis.</div>}
        </>
    );
};

export default LoginBox;
